# significant notation
# single value
sigStar =function(x){
  if (x > 0.05){ star = " "} else{
    if (x>= 0.01) { star = "*"} else {
      if (x >= 0.001) { star = "**"} else { 
        if (x >= 0.0001) { star = "***" } else {
          star = "****"
        }  
      }
    }
  }
  return(star)
}

# multiple values in a single vector
sigStarS = function(vector){
  output = c()
  for (a in vector) {
    output = append(output, sigStar(a))
  }
  return(output)
}
